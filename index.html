struct Royalty : public ContractBase {
  id wallets[5];
  void distribute(uint64 amt) {
    uint64 share = amt / 5;
    for (int i=0;i<4;i++) pay(wallets[i],share);
    pay(wallets[4], amt - share*4);
  }
};
